# Agrega las URLs que quieres monitorear, una por línea
# Puedes usar filtros CSS o XPath para partes específicas

# Ejemplo: Monitorear título de página
name: "Ejemplo - Título principal"
url: https://example.com
filter:
  - css: h1

---

# Ejemplo: Monitorear precio de producto
name: "Producto X - Precio"
url: https://tienda.com/producto
filter:
  - css: .precio
  - strip

---

# Ejemplo: Monitorear página completa
https://ejemplo.com/noticias

---
